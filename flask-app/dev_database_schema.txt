
CREATE TABLE courses (
  code VARCHAR(6) UNIQUE PRIMARY KEY,
  name VARCHAR(30),
  course_page VARCHAR(50),
  type VARCHAR(10)
);

CREATE TABLE course_periods
(
    code VARCHAR(6),
    period VARCHAR(3),
    PRIMARY KEY (code, period),
    FOREIGN KEY (code) REFERENCES courses
);

CREATE TABLE course_blocks
(
    code VARCHAR(6),
    block VARCHAR(1),
    PRIMARY KEY (code, block),
    FOREIGN KEY (code) REFERENCES courses
);
